<script type="module">

  import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.1/firebase-app.js";

  import { getFirestore, collection, addDoc, doc, updateDoc, deleteDoc, onSnapshot, query, orderBy } from "https://www.gstatic.com/firebasejs/10.12.1/firebase-firestore.js";


  // 已经为你填好了专属云端钥匙

  const firebaseConfig = {

    apiKey: "AIzaSyB_e-0wi3aW2pqWOuZFeLbTe5JJ07UrIGU",

    authDomain: "mos-album.firebaseapp.com",

    projectId: "mos-album",

    storageBucket: "mos-album.firebasestorage.app",

    messagingSenderId: "194609146265",

    appId: "1:194609146265:web:62d79bc4880d7ff6f3cb08"

  };


  const app = initializeApp(firebaseConfig);

  const db = getFirestore(app);

  const postsCollection = collection(db, "moments");


  // --- 基础工具库 ---

  window.showToast = (msg) => {

    const t = document.getElementById('toast');

    t.textContent = msg; t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 3000);

  };

  

  window.closeModal = (id) => document.getElementById(id).classList.remove('active');


  async function compressImage(file) {

    return new Promise((resolve) => {

      const reader = new FileReader(); reader.readAsDataURL(file);

      reader.onload = (event) => {

        const img = new Image(); img.src = event.target.result;

        img.onload = () => {

          const canvas = document.createElement('canvas');

          let w = img.width, h = img.height;

          if (w > 800) { h = Math.round((h * 800) / w); w = 800; }

          canvas.width = w; canvas.height = h;

          canvas.getContext('2d').drawImage(img, 0, 0, w, h);

          resolve(canvas.toDataURL('image/jpeg', 0.6));

        };

      };

    });

  }


  // --- 个人资料管理 ---

  let userProfile = JSON.parse(localStorage.getItem('mos_profile')) || {

    name: "Qinyi",

    avatar: "https://api.dicebear.com/7.x/avataaars/svg?seed=Qinyi",

    cover: "https://images.unsplash.com/photo-1464822759023-fed622ff2c3b?w=1200"

  };

  let tempAvatar = userProfile.avatar;


  function renderProfile() {

    document.getElementById('display-name').textContent = userProfile.name;

    document.getElementById('display-avatar').src = userProfile.avatar;

    document.getElementById('display-cover').style.backgroundImage = `url('${userProfile.cover}')`;

  }

  renderProfile();


  window.openProfileModal = () => {

    document.getElementById('edit-name-input').value = userProfile.name;

    document.getElementById('edit-avatar-preview').src = userProfile.avatar;

    tempAvatar = userProfile.avatar;

    document.getElementById('profile-modal').classList.add('active');

  };


  document.getElementById('avatar-upload').addEventListener('change', async (e) => {

    if(e.target.files[0]) {

      tempAvatar = await compressImage(e.target.files[0]);

      document.getElementById('edit-avatar-preview').src = tempAvatar;

    }

  });


  document.getElementById('cover-upload').addEventListener('change', async (e) => {

    if(e.target.files[0]) {

      userProfile.cover = await compressImage(e.target.files[0]);

      localStorage.setItem('mos_profile', JSON.stringify(userProfile));

      renderProfile();

      showToast("封面已更新");

    }

  });


  window.saveProfile = () => {

    userProfile.name = document.getElementById('edit-name-input').value;

    userProfile.avatar = tempAvatar;

    localStorage.setItem('mos_profile', JSON.stringify(userProfile));

    renderProfile();

    closeModal('profile-modal');

  };


  // --- 云端动态管理 ---

  let currentImages = [];

  let postsList = [];


  const q = query(postsCollection, orderBy("time", "desc"));

  onSnapshot(q, (snapshot) => {

    postsList = [];

    snapshot.forEach((doc) => {

      postsList.push({ id: doc.id, ...doc.data() });

    });

    renderFeed();

  });


  function renderFeed() {

    const container = document.getElementById('feed-container');

    if (postsList.length === 0) {

      container.innerHTML = '<div style="text-align: center; color: #999; margin-top: 50px;">点击右下角加号，发布第一条动态吧！</div>';

      return;

    }

    

    container.innerHTML = '';

    postsList.forEach(post => {

      let grid = 'grid-9'; 

      if (post.images.length === 1) grid = 'grid-1'; 

      else if ([2, 4].includes(post.images.length)) grid = 'grid-4';

      

      const timeStr = new Date(post.time).toLocaleString('zh-CN', { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' });


      // 在这里加上了修改和删除两个按钮

      container.innerHTML += `

        <div class="post">

          <img class="avatar" src="${post.authorAvatar}">

          <div class="post-body">

            <div class="author-name">${post.authorName}</div>

            <div class="post-text">${post.text}</div>

            ${post.images && post.images.length ? `<div class="img-grid ${grid}">${post.images.map(img => `<img src="${img}">`).join('')}</div>` : ''}

            ${post.location ? `<div class="location-tag">${post.location}</div>` : ''}

            <div class="meta-info">

              <span>${timeStr}</span>

              <div class="action-btns">

                <button class="action-btn btn-edit" onclick="openPostModal('edit', '${post.id}')">修改</button>

                <button class="action-btn btn-delete" onclick="deletePost('${post.id}')">删除</button>

              </div>

            </div>

          </div>

        </div>`;

    });

  }


  function updateImagePreviews() {

    const area = document.getElementById('image-preview-area');

    area.querySelectorAll('.preview-img-box').forEach(el => el.remove());

    currentImages.forEach((img, i) => {

      const box = document.createElement('div'); box.className = 'preview-img-box';

      box.innerHTML = `<img src="${img}"><div class="remove-img" onclick="removeImage(${i})">X</div>`;

      area.insertBefore(box, document.getElementById('add-img-btn-label'));

    });

  }

  

  window.removeImage = (index) => { currentImages.splice(index, 1); updateImagePreviews(); };


  document.getElementById('post-images-upload').addEventListener('change', async (e) => {

    for (let file of Array.from(e.target.files).slice(0, 9 - currentImages.length)) {

      currentImages.push(await compressImage(file));

    }

    updateImagePreviews();

  });


  window.openPostModal = (mode, id = null) => {

    const modal = document.getElementById('post-modal');

    modal.classList.add('active');

    

    if (mode === 'create') {

      document.getElementById('modal-title').textContent = '记录这一刻';

      document.getElementById('post-id').value = '';

      document.getElementById('post-text').value = '';

      document.getElementById('post-location').value = '';

      

      const now = new Date();

      now.setMinutes(now.getMinutes() - now.getTimezoneOffset());

      document.getElementById('post-time').value = now.toISOString().slice(0, 16);

      currentImages = [];

    } else {

      document.getElementById('modal-title').textContent = '修改动态';

      const post = postsList.find(p => p.id === id);

      document.getElementById('post-id').value = post.id;

      document.getElementById('post-text').value = post.text;

      document.getElementById('post-location').value = post.location;

      document.getElementById('post-time').value = post.time;

      currentImages = post.images ? [...post.images] : [];

    }

    updateImagePreviews();

  };


  window.savePost = async () => {

    const id = document.getElementById('post-id').value;

    const postData = {

      text: document.getElementById('post-text').value,

      location: document.getElementById('post-location').value,

      time: document.getElementById('post-time').value,

      images: currentImages,

      authorName: userProfile.name,

      authorAvatar: userProfile.avatar

    };


    try {

      if (id) {

        const postRef = doc(db, "moments", id);

        await updateDoc(postRef, postData);

        showToast("修改已同步至云端");

      } else {

        await addDoc(postsCollection, postData);

        showToast("发布成功");

      }

      closeModal('post-modal');

    } catch (e) {

      console.error(e);

      showToast("保存失败，请检查网络");

    }

  };


  // 完整的删除功能在这里

  window.deletePost = async (id) => {

    if(confirm("确定要删除这条动态吗？删除后不可恢复。")) {

      try {

        await deleteDoc(doc(db, "moments", id));

        showToast("动态已删除");

      } catch (e) {

        console.error(e);

        showToast("删除失败，请稍后再试");

      }

    }

  };
